@startuml
package Model {
    class Warehouse {
        id : uint
        floors : vector<Floor>
    }

    class Floor {
        id : uint
        goodsCollections : vector<GoodsCollection>
    }

    class GoodsCollection {
        id : uint
        goods : vector<Good>
        quantity : uint
        temperature : float
    }

    class Good {
        goodInformation : GoodInformation
    }

    class GoodInformation {
        name : string
        weight : float
    }
    
    class Truck
    {
        pos : float2
        shouldMoveTo : float2
        truckInformation : TruckInformation
    }
    
    class TruckInformation
    {
        dateOfLatestService : string
        truckManufactor : string
        batteryType : string
    }
}

package View {
    class GoodView
    class WarehouseView
    class GoodsCollectionView
    class StartScreen
    class TruckView
}

package Controller {
    class WarehouseController
    class GoodsCollectionController
    class GoodController
    class TruckController
}
@enduml
